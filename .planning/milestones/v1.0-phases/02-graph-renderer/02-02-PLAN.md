interface FileGraph {
  readonly files: ReadonlySet<string>;
  readonly forwardEdges: ReadonlyMap<string, ReadonlySet<string>>;
  readonly reverseEdges: ReadonlyMap<string, ReadonlySet<string>>;
  readonly definitions: ReadonlyMap<string, readonly CodeNode[]>;
}

interface PageRankOptions {
  readonly alpha: number;          // damping factor, default 0.85
  readonly maxIterations: number;  // default 100
  readonly tolerance: number;      // convergence delta, default 1e-6
  readonly personalization?: ReadonlyMap<string, number>;
}

interface PageRankResult {
  readonly scores: ReadonlyMap<string, number>;
  readonly iterations: number;
  readonly converged: boolean;
}

<task type="auto" tdd="true">
  <name>Task 1: Standard PageRank with power iteration</name>
  <files>src/graph/pagerank.ts, tests/graph/pagerank.test.ts</files>
  <behavior>
    - Test: Empty graph (0 files) returns empty Map, 0 iterations, converged=true
    - Test: Single file with no edges returns score 1.0 for that file
    - Test: Linear chain A->B->C: C scores highest (most "imported"), A scores lowest
    - Test: Star pattern (A->X, B->X, C->X, D->X): X scores highest — hub detection
    - Test: All scores sum to approximately 1.0 (within 0.01 tolerance)
    - Test: No single file monopolizes rank (max score < 0.9 for graphs with >2 nodes)
    - Test: Dangling node (file with inbound edges but no outbound) does NOT get score ~1.0
    - Test: Convergence happens within 100 iterations for typical small graphs
    - Test: Two disconnected subgraphs each have internally consistent rankings
  </behavior>
  <action>
    1. Create `src/graph/pagerank.ts` with:
       - `computePageRank(graph: FileGraph, options: PageRankOptions): PageRankResult`
       - Internal `buildTeleportVector(nodes: readonly string[], personalization: ReadonlyMap<string, number> | undefined): Map<string, number>`
       - Algorithm (power iteration per research Pattern 3):
         a. Extract nodes array from graph.files
         b. Handle N=0 edge case (return empty)
         c. Build teleport vector (uniform 1/N if no personalization)
         d. Initialize all scores to 1/N
         e. Precompute outDegree for each node from forwardEdges
         f. Per iteration: compute danglingSum (sum of scores for nodes with outDegree=0), then for each node v: newScore = alpha * (linkRank + danglingSum * teleport[v]) + (1-alpha) * teleport[v]
         g. Compute L1 delta between new and old scores
         h. If delta < tolerance, break and mark converged=true
         i. Return PageRankResult with final scores, iteration count, converged flag

    2. Create `tests/graph/pagerank.test.ts` with TDD tests per behavior spec.
       - Build test FileGraph objects directly (construct files Set, forwardEdges/reverseEdges Maps manually — do NOT use buildGraph, which is Plan 01 territory)
       - Helper function `makeGraph(edges: [string, string][]): FileGraph` that builds from edge pairs
       - Include `assertConverged` helper: checks sum ~1.0, no single score >0.9
  </action>
  <verify>
    <automated>pnpm vitest run tests/graph/pagerank.test.ts</automated>
  </verify>
  <done>Standard PageRank converges correctly. Hub files rank higher than leaf files. Dangling nodes redistribute rank. Scores sum to ~1.0. All tests green.</done>
</task>
